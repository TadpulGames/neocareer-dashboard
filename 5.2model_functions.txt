export async function GetResult() {
    StartResultTimer();
    const jobName = jobNameRoot;
    const email = emailRoot_firebaseSafe;

    const evaluationFunction = generateEvaluationFunctionFromFirebase(jobName, cachedJobData.Metrics);
    const response = await OpenAIRequestResult(modifiedResultContext, evaluationFunction);

    await Analytics_TookTimeForResult();

    // 🧩 FIX: response.arguments is already parsed JSON
    let evaluation = response.arguments;

      // ✅ Fix: handle stringified JSON
  if (typeof evaluation === "string") {
    try {
      evaluation = JSON.parse(evaluation);
      // console.log("✅ Parsed stringified JSON successfully");
    } catch (err) {
      console.error("❌ Failed to parse JSON:", err);
    }
  }

// 🧩 Handle possible nested structure
const parsedEval = evaluation?.criteria_breakdown 
  ? evaluation 
  : evaluation?.evaluation 
    ? evaluation.evaluation 
    : null;

if (!parsedEval || !parsedEval.criteria_breakdown || !parsedEval.criteria_scores) {
  console.error("❌ Invalid evaluation format:", evaluation);
  throw new Error("Evaluation result missing criteria_breakdown or criteria_scores");
}

        const Results = [];
        let totalScore = 0;
        let criteriaCount = 0;
        const criteriaBreakdown = parsedEval.criteria_breakdown;
        const criteriaScores = parsedEval.criteria_scores;

        for (const [metricName, feedback] of Object.entries(criteriaBreakdown)) {
           const score = criteriaScores?.[metricName] ?? 0; 

            Results.push({
                Feedback: feedback,
                MetricName: metricName,
                score: score
            });

            totalScore += score;
            criteriaCount++;
        }

        const overallScore = criteriaCount > 0 ? totalScore / criteriaCount : 0;
      
        const resultDocRef = doc(dbRef, companyNameRoot, FIREBASEDOCUMENT.JOBS, jobName, FIREBASESUBDOCUMENT.CANDIDATES, email, FIREBASESUBSUBDOCUMENT.RESULT);
       
   
        const resultDocSnap = await getDoc(resultDocRef);
       
        const shouldUploadResults = !resultDocSnap.exists() ||
            !resultDocSnap.data() ||
            !resultDocSnap.data().Results ||
            resultDocSnap.data().Results.length === 0;

        if (shouldUploadResults) {
             await setDoc(resultDocRef, { Results }, { merge: true });

            const statusRef = doc(dbRef, companyNameRoot, FIREBASEDOCUMENT.JOBS, jobName, FIREBASESUBDOCUMENT.CANDIDATES, email, FIREBASESUBSUBDOCUMENT.STATUS);

            await setDoc(statusRef, { isInterviewFinished: "True" }, { merge: true });

            const tempConversationRef = doc(dbRef, companyNameRoot, FIREBASEDOCUMENT.JOBS, jobNameRoot, FIREBASESUBDOCUMENT.CANDIDATES, email, FIREBASESUBSUBDOCUMENT.TEMPCONVERSATION);
            const tempConversationDocSnap = await getDoc(tempConversationRef);

            if (tempConversationDocSnap.exists()) {
                const conversationRef = doc(dbRef, companyNameRoot, FIREBASEDOCUMENT.JOBS, jobNameRoot, FIREBASESUBDOCUMENT.CANDIDATES, email, FIREBASESUBSUBDOCUMENT.CONVERSATION);
                await setDoc(conversationRef, tempConversationDocSnap.data());
                await deleteDoc(tempConversationRef);
            }
            
            // Get existing CandidateList entry to preserve terminatedForViolations flag
            const candidateListRef = doc(dbRef, companyNameRoot, FIREBASEDOCUMENT.CANDIDATELIST, "List", jobName);
            const candidateListSnap = await getDoc(candidateListRef);
            let existingTerminationFlag = false;

            if (candidateListSnap.exists()) {
                const data = candidateListSnap.data();
                if (data[email] && Array.isArray(data[email]) && data[email].length > 0) {
                    existingTerminationFlag = data[email][0].terminatedForViolations || false;
                }
            }

            userNameRoot = sessionStorage.getItem("candidateName");
 
            CandidateData.CandidateName = userNameRoot;
            CandidateData.InterviewDate = getCurrentTimeFormat();
            CandidateData.OverallScore = overallScore;
            CandidateData.Results = Results;
            CandidateData.candidateStatusCode = 0;

            // Preserve terminatedForViolations flag if it was set
            if (existingTerminationFlag) {
                CandidateData.terminatedForViolations = true;
            }

            await updateDoc(candidateListRef, { [email]: deleteField() });

            await setDoc(candidateListRef, { [email]: [CandidateData] }, { merge: true });
            // console.log("✅ CandidateList updated successfully with Results (terminatedForViolations preserved):");
                await UpdateCredits(jobName);

    }
}

function getModelForRequest(mode, extraParams = {}) {
  // Deep reasoning / evaluation / structured output
  if (
    extraParams?.functions ||
    extraParams?.function_call ||
    extraParams?.response_format
  ) {
    return "gpt-5.2";
  }

  // Normal conversation
  if (mode === "chat") {
    return "gpt-5.2-chat-latest";
  }

  return "gpt-5.2-chat-latest";
}
